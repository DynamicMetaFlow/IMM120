// Variables
var x = 100;
var y = 100;
var w = 200;
var h = 200;

function setup() {
  createCanvas(400, 400); // size
}

function draw() {
  background(255);
  fill(255);
  if(mouseX > x && mouseX < x+w && mouseY > y && mouseY < y+h) {
    fill(0);
  }
  if(isMousePressed) {
    line(y+h, 0, y+h, height);
  }
  rect(x, y, w, h);
}
