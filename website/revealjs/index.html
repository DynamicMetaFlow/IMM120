var eyes = [];

function setup() {
  createCanvas(640, 360);

  eyes.push(new Eye(250,  16, 120));
  eyes.push(new Eye(164, 185,  80));
  eyes.push(new Eye(410, 230, 220));
  eyes.push(new Eye(-300, 100, 800));
  eyes.push(new Eye(220, height-40, 50));
  eyes.push(new Eye(400, 40, 60));
  eyes.push(new Eye(width-50, height-80, 110));
  eyes.push(new Eye(width-80, 80, 80));

  strokeWeight(16);
}

function draw() {
  background(102, 50);

  noStroke();
  for(var i=0;i<eyes.length;i++) {
    eyes[i].update(mouseX, mouseY);
    eyes[i].display();
  }

  stroke(0, 204, 153);
  line(mouseX, mouseY, pmouseX, pmouseY);
}

function Eye(tx, ty, ts) {
  this.x = tx;
  this.y = ty;
  this.size = ts;
  this.angle = 0.0;
}
Eye.prototype.update = function(mx, my) {
  this.angle = atan2(my-this.y, mx-this.x);
};
Eye.prototype.display = function() {
  push();
  translate(this.x, this.y);
  if (mouseIsPressed) {
    fill(51);
    ellipse(0, 0, this.size, this.size);
  } else {
    fill(255);
    ellipse(0, 0, this.size, this.size);
    rotate(this.angle);
    fill(153, 204, 0);
    ellipse(this.size/4, 0, this.size/2, this.size/2);
  }
  pop();
}
